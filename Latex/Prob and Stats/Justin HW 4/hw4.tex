\begin{equation}
    s_x{}_y^2 \approx \overline{xy} - \bar{x}\bar{y}
\end{equation}

then this can be rewritten as the following: 
\begin{equation}
   s_x{}y^2 \approx  \sum\limits_{i=0}^{i-1} \frac{x_i yi}{N} - \sum\limits_{i=0}^{i-1} \frac{xi}{N} \sum\limits_{i=0}^{i-1} \frac{y_i}{N}
\end{equation}
